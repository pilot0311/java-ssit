*** 16. 이름, 나이, 국어, 영어, 수학을 입력받아서

    아래와 같이 출력
    [입력형식] 이름 나이 국어 영어 수학 입력 ? 
    [출력형식] "홍길동" 20살 10 20 30 60 20.00   
    
    ( 조건 : Scanner 클래스 사용 )   
    ( 조건 : 평균은 소수점 2자리까지 출력 )   
    package days03;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.Scanner;

public class Ex01 {
	
	public static void main(String[] args) throws IOException {
		String name;
		byte age;
		byte kor;
		byte eng;
		byte math;
		short total;
		double avg;
		Scanner sc = new Scanner(System.in);
		System.out.printf("이름 나이 국어 영어 수학 입력");
		name = sc.next();
		age = sc.nextByte();
		kor = sc.nextByte();
		eng = sc.nextByte();
		math = sc.nextByte();
		total = (short)(kor+eng+math);
		avg = (double)total/3;
		System.out.printf("\"%s\" %d살 %d %d %d %d %.2f",name,age,kor,eng,math,total,avg);
	}

}
